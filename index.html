Config

SignConfig


# ----------------------------------------------------------
#             CommandSigns | Plugin made by GonzaDL
# ----------------------------------------------------------
# Here, you could configured the messages of the signs!
# Remember when you finish, put /csreload, For load the new settings!
# ----------------------------------------------------------
#                         Settings
# ----------------------------------------------------------
# Sign (1)
    set {Sign1Name} to "[CS]"
    set {Sign1Command} to "/Cmd"
    set {Sign1Line1} to "§8[§cCommandSigns§8]"
    set {Sign1Line2} to "§7Plugin created by"
    set {Sign1Line3} to "§7GonzaDL"
    set {Sign1Line4} to "§7-----------"
# ----------------------------------------------------------
# Sign (2)
    set {Sign2Name} to "[CS1]"
    set {Sign2Command} to "/Cmd"
    set {Sign2Line1} to "§8[§cCommandSigns§8]"
    set {Sign2Line2} to "§7Plugin created by"
    set {Sign2Line3} to "§7GonzaDL"
    set {Sign2Line4} to "§7-----------"
# ----------------------------------------------------------
